== https://nodejs.org[node]

.zsh
----
$ brew install nvm
$ mkdir -p $HOME/.nvm
$ cat >> $HOME/.zshrc <<'EOF'
export NVM_DIR="$HOME/.nvm"
[ -s "/opt/homebrew/opt/nvm/nvm.sh" ] && . "/opt/homebrew/opt/nvm/nvm.sh"
[ -s "/opt/homebrew/opt/nvm/etc/bash_completion.d/nvm" ] && . "/opt/homebrew/opt/nvm/etc/bash_completion.d/nvm"
EOF
$ source $HOME/.zshrc
----
